*Note: The synthesizer only works on macOS and Linux due to WSL permission errors with audio

# macOS (homebrew)

brew install libao
opam install mm
opam install ao
opam install graphics

# Ubuntu/WSL

# install mm library
opam install mm

# install ao
sudo apt-get install libao-dev
opam install ao

# install graphics (if not yet installed)
opam install graphics
